CREATE TABLE "Schedule" ("Id" INTEGER PRIMARY KEY NOT NULL, "Task_id" INTEGER NOT NULL REFERENCES "Task" ("Id"), "Room_id" INTEGER NOT NULL REFERENCES "Room" ("Id"), "Teacher_id" INTEGER NOT NULL REFERENCES "Teacher" ("Id"), "Class_id" INTEGER NOT NULL REFERENCES "Class" ("Id"), "Task_time_id" INTEGER NOT NULL REFERENCES "Task_time" ("Id"), "Day" DATE NOT NULL);
CREATE TABLE "Room" ("Id" INTEGER PRIMARY KEY NOT NULL, "Name" VARCHAR(10), "address" VARCHAR(255));
CREATE TABLE "Task_type" ("Id" INTEGER PRIMARY KEY NOT NULL, "Name" VARCHAR(255) NOT NULL);
CREATE TABLE "Task_time" ("Id" INTEGER PRIMARY KEY NOT NULL, "StartTime" VARCHAR(10) NOT NULL, "EndTime" VARCHAR(10));
CREATE TABLE "Task" ("Id" INTEGER PRIMARY KEY NOT NULL, "Task_type_id" INTEGER NOT NULL REFERENCES "Task_type" ("Id"), "Name" VARCHAR(255) NOT NULL);
CREATE TABLE "Class" ("Id" INTEGER PRIMARY KEY NOT NULL, "Name" VARCHAR(255) NOT NULL, "Faculty" VARCHAR(255), "EduOrg" VARCHAR(255), "City" VARCHAR(255), "HasSchedule" NUMERIC(1,0) DEFAULT 0, "LastUpdate" DATE);
CREATE TABLE "Teacher" ("Id" INTEGER PRIMARY KEY NOT NULL, "Name" VARCHAR(255) NOT NULL, "EduOrg" VARCHAR(255), "City" VARCHAR(255), "HasSchedule" NUMERIC(1,0) DEFAULT 0, "LastUpdate" DATE DEFAULT CURRENT_TIMESTAMP);
